## Except And Intersect

#Except (차집합)

SELECT Column1 FROM Table1
	EXCEPT SELECT Column1 FROM Table2

#Intersect(교집합)

SELECT Column1 FROM Table1
	INTERSECT SELECT Column1 FROM Table2
